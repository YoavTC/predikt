using UnityEngine;

public class BoardInitializer : MonoBehaviour
{
    public Cell cellPrefab; // Assign a prefab for the cell in the inspector
    public int boardSize = 9; // Size of the board (default 9x9)
    public float cellSize = 1.0f; // Size of each cell (default 1 unit)
    public float yOffset = 0.1f; // Offset for the Y axis

    public Cell[,] boardCells; // Store references to all cells
    
    private void Start()
    {
        GenerateBoard();
    }

    private void GenerateBoard()
    {
        if (cellPrefab == null)
        {
            Debug.LogError("Cell prefab is not assigned!");
            return;
        }

        boardCells = new Cell[boardSize, boardSize];

        for (int x = 0; x < boardSize; x++)
        {
            for (int y = 0; y < boardSize; y++)
            {
                Vector3 position = new Vector3(x * cellSize, y * yOffset, y * cellSize);
                
                Cell cell = Instantiate(cellPrefab, position, Quaternion.identity).GetComponent<Cell>();
                
                cell.transform.SetParent(transform);
                cell.name = $"Cell_{x}_{y}";

                cell.Init(x, y);
                
                boardCells[x, y] = cell;
            }
        }

        Debug.Log("Board generated successfully.");
    }
}
